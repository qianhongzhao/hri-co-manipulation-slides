\begin{frame}{Reference task: Collaborative table carrying}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Task in Diffusion Co-Policy}
\begin{itemize}
  \item A human and a robot act at opposite ends of a rigid table.
  \item They move the table to a goal while avoiding obstacles.
  \item Each partner applies a planar force; success requires coordinated translation and rotation.
\end{itemize}
\end{block}
\column{0.48\textwidth}
\begin{block}{Information available online}
\begin{itemize}
  \item table pose and velocity;
  \item initial pose, goal, and obstacle locations;
  \item recent human force commands and robot actions;
  \item for physical co-manipulation: add measured 6D interaction wrench and human motion.
\end{itemize}
\end{block}
\end{columns}
\vspace{0.18cm}
\begin{alertblock}{Why human intention must be modeled}
The same table state can lead to different valid routes, rotations, and leader--follower roles. Without predicting the human's choice, the robot may resist the human, apply conflicting forces, or react too late.
\end{alertblock}
\end{frame}

\begin{frame}{Human intention is hidden but control-relevant}
\begin{columns}[T,onlytextwidth]
  \column{0.50\textwidth}
  \begin{block}{What the robot observes}
  \begin{itemize}
    \item object pose and velocity $x_t^O$;
    \item interaction wrench $\mathbf{w}_t^{\mathrm{int}}$ (3D force + 3D torque);
    \item human/robot motion and recent actions;
    \item task geometry, goal, and constraints $c$.
  \end{itemize}
  \end{block}
  \column{0.46\textwidth}
  \begin{block}{What remains hidden}
  \begin{itemize}
    \item desired object motion or local goal;
    \item leader/follower role and compliance;
    \item effort, force, and safety preferences;
    \item responses to possible robot actions.
  \end{itemize}
  \end{block}
\end{columns}
\vspace{0.25cm}
\begin{alertblock}{Central question}
How should human intention be represented and inferred so that it improves real-time robot planning and control in physically coupled tasks?
\end{alertblock}
\end{frame}

\begin{frame}{A useful model must resolve four uncertainties}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{itemize}
  \item \key{Behavioral multimodality}: the same state may admit several valid goals, paths, or roles.
  \item \key{Partial observability}: motion and force provide indirect, sometimes conflicting evidence.
\end{itemize}
\column{0.48\textwidth}
\begin{itemize}
  \item \key{Closed-loop interaction}: the human responds to the robot and may change strategy.
  \item \key{Bounded rationality}: actions may be delayed, noisy, effort-limited, or non-optimal.
\end{itemize}
\end{columns}
\vspace{0.35cm}
\begin{block}{Terminology}
``Multimodal sensing'' means multiple input types (motion, wrench, vision). 

``Multimodal behavior'' means multiple possible future human strategies.
\end{block}
\end{frame}

\begin{frame}[shrink=5]{Shared papers show human models benefit interaction}
\small
\begin{columns}[T,onlytextwidth]
\column{0.315\textwidth}
\begin{block}{Diffusion Co-Policy}
\begin{itemize}
  \item Generates several possible joint force/action sequences
  \item Demonstrates adaptation and role switching
  \item Provides the physical test setting for collaborative transport
\end{itemize}
\end{block}
\column{0.315\textwidth}
\begin{block}{Bounded-Rational Model}
\begin{itemize}
  \item Shows joint human actions can be represented by a game model
  \item Bounded rationality fits observed behavior better
  \item Leadership and adaptation affect performance
\end{itemize}
\end{block}
\column{0.315\textwidth}
\begin{block}{PACE / N-PACE}
\begin{itemize}
  \item Estimates intent/cost parameters online
  \item Treats the peer as another adapting learner
  \item Shows a reasonable peer model can improve interaction policies
\end{itemize}
\end{block}
\end{columns}
\vspace{0.25cm}
\begin{alertblock}{Combined insight}
Human intention is modelable and useful for control; the unresolved question is which representation is sufficiently expressive, identifiable, and transferable.
\end{alertblock}
\end{frame}

\begin{frame}{A hierarchy of control-relevant intention representations}
\begin{block}{High-level layer: interaction primitive}
Output: $z_H=(m,g,v,\rho)$, describing the current phase, local goal, speed/force level, and interaction role.
\end{block}
\centering\textcolor{orange}{\Large$\Downarrow$}
\vspace{0.08cm}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Desired object motion}
Terminal pose, path mode, or desired trajectory $\xi_H^*$.
\end{block}
\begin{block}{Cost and preferences}
Features $\phi$ and weights $w_H$ for progress, effort, force, and safety.
\end{block}
\column{0.48\textwidth}
\begin{block}{Local dynamical system}
Human-intended vector field $f_H(x^O;\theta_H)$ over object motion.
\end{block}
\begin{block}{Interactive response distribution}
Human wrench/action responses conditioned on candidate robot actions.
\end{block}
\end{columns}
\end{frame}

\begin{frame}{How the intention representations will be evaluated}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{1. Intention identifiability}
Can the model distinguish route, role, and task phase early? Measure classification accuracy, posterior confidence, and detection delay.
\end{block}
\begin{block}{2. Future prediction}
Measure trajectory displacement, wrench prediction error, likelihood of observed behavior, and coverage of different valid futures.
\end{block}
\column{0.48\textwidth}
\begin{block}{3. Closed-loop control value}
Measure task success, completion time, human effort, conflicting interaction force, safety violations, and online computation time.
\end{block}
\begin{block}{4. Transfer and adaptation}
Test unseen users, objects, obstacle layouts, and tasks; report zero-shot performance and data required for personalization.
\end{block}
\end{columns}
\end{frame}

\begin{frame}{High-level option: Reusable interaction primitives}
\begin{columns}[T,onlytextwidth]
\column{0.52\textwidth}
\begin{block}{Switching latent-state model}
\[
z_t^H=(m_t,g_t,v_t,\rho_t),\qquad m_t\in\mathcal M_{\mathrm{int}}
\]
where $\mathcal M_{\mathrm{int}}$ includes translate, rotate, hold, align, insert, and yield.
\begin{itemize}
  \item $m_t$: current interaction primitive;
  \item $g_t$: local direction, pose, or contact goal;
  \item $v_t$: desired speed or interaction-force level;
  \item $\rho_t$: leader/follower role or compliance.
\end{itemize}
\end{block}
\column{0.44\textwidth}
\begin{exampleblock}{Why it supports generalization}
Different tasks reuse a small vocabulary: transport may be \emph{translate--rotate--align}; insertion may be \emph{approach--align--hold--insert}.
\end{exampleblock}
\begin{alertblock}{Initial primitive vocabulary}
Start with six object-level modes: translate, rotate, hold, align, insert, and yield. They are defined by distinct motion/force patterns, not by a specific task name.
\end{alertblock}
\end{columns}
\end{frame}

\begin{frame}{How primitives are identified and learned}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Identify the current primitive}
\begin{itemize}
  \item Use task-phase labels when the phase is known.
  \item Otherwise use Expectation--Maximization (EM) to infer latent switching modes from object motion, wrench, and context.
  \item Infer $p(m_t\mid h_t,c)$ online rather than committing to one mode too early.
\end{itemize}
\end{block}
\column{0.48\textwidth}
\begin{block}{Decide how many primitives are needed}
Begin with the six-mode vocabulary. Merge modes with indistinguishable motion/force responses; add a mode only when repeated residual patterns improve held-out prediction and control.
\end{block}
\begin{alertblock}{Learn continuous parameters}
Within each mode, train goal, speed/force, and role parameters from future object motion and interaction-wrench prediction.
\end{alertblock}
\end{columns}
\end{frame}

\begin{frame}{Option A: Desired object motion as intention}
\begin{columns}[T,onlytextwidth]
\column{0.52\textwidth}
\begin{block}{Desired trajectory model}
\[
p_\psi(\xi_H^*\mid h_t,c),\qquad
\xi_H^*=\{x_{O,k}^{H,*}\}_{k=t}^{t+H}
\]
$h_t$ contains recent object/human/robot motion and interaction wrench; $c$ contains task goal, geometry, object, and constraints.
\end{block}
\textbf{Possible outputs}
\begin{itemize}
  \item terminal pose or waypoint;
  \item discrete path/strategy mode;
  \item distribution over full desired trajectories.
\end{itemize}
\column{0.44\textwidth}
\begin{exampleblock}{Why it is attractive}
Directly useful as a reference for shared-object MPC; easier to label and visualize than latent cost parameters.
\end{exampleblock}
\begin{alertblock}{Main challenges}
Multiple valid trajectories; changing intention; passive prediction ignores robot influence.
\end{alertblock}
\end{columns}
\end{frame}

\begin{frame}{Option A: Training and control interface}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Training without a unique intent label}
Use future object motion as a self-supervised target, but learn several possible future trajectories rather than one ``correct'' path. Controlled experiments can also provide route or goal labels.
\end{block}
\column{0.48\textwidth}
\begin{block}{Learning objective}
Use a diffusion loss or best-of-$K$ prediction loss so that left/right routes and alternative rotation strategies are retained instead of averaged together.
\end{block}
\end{columns}
\vspace{0.2cm}
\begin{alertblock}{Role in the complete system}
The desired motion is a low-level reference for shared-object model predictive control (MPC). Combine it with a primitive or preference model when the robot must explain why the human changes strategy.
\end{alertblock}
\vspace{0.12cm}
Related: Townsend et al. (2017); Franceschi et al. (2023).
\end{frame}

\begin{frame}{Option B: Infer what the human is optimizing}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.53\textwidth}
\begin{block}{Task-conditioned latent cost}
\vspace{-0.12cm}
\[
J_H=\sum_{k=t}^{t+H}{w_k^H}^{\!\top}
\phi_\psi(x_k^O,u_k^H,u_k^R,c)
\]
\vspace{-0.12cm}
\begin{itemize}
  \item $\phi_\psi$: shared learned and physical features for progress, effort, force, safety, and coordination;
  \item $w_H$: person-specific preference weights inferred online.
\end{itemize}
\end{block}
\column{0.43\textwidth}
\begin{exampleblock}{Why it is attractive}
Interpretable tradeoffs and a direct connection to game-theoretic planning, IOC, PACE, and N-PACE.
\end{exampleblock}
\begin{alertblock}{Main challenges}
Non-identifiability, semantic drift in learned features, and approximate human optimality.
\end{alertblock}
\end{columns}
\vspace{0.08cm}
\begin{block}{Training without cost labels}
Learn shared $\phi_\psi$ from demonstrated actions via max-entropy IOC/IRL with physical-feature anchoring; infer person-specific $w_H$ as a latent online variable. No direct cost labels are required.
\end{block}
\end{frame}

\begin{frame}{The trained cost model can become N-PACE's human model}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Offline learned representation}
\begin{itemize}
  \item Train a shared task-conditioned feature map $\phi_\psi$ from multi-person demonstrations.
  \item Learn a context-dependent prior $p(w_H\mid c)$.
  \item Freeze $\phi_\psi$ at deployment so the online parameter meaning remains stable.
\end{itemize}
\end{block}
\column{0.48\textwidth}
\begin{block}{Online N-PACE layer}
\begin{itemize}
  \item Define $J_H=\sum_k w_H^\top\phi_\psi(x_k,u_k^H,u_k^R,c)$.
  \item N-PACE estimates $w_H$ and how the human is updating beliefs about the robot.
  \item The inferred cost enters the dynamic-game solver to update robot actions.
\end{itemize}
\end{block}
\end{columns}
\vspace{0.25cm}
\begin{alertblock}{What the learned model contributes}
It replaces task-by-task manual feature design with a reusable cost representation; N-PACE retains peer-aware online inference and interaction-aware control.
\end{alertblock}
\end{frame}

\begin{frame}{Option C: Intention as a local vector field}
\begin{columns}[T,onlytextwidth]
\column{0.52\textwidth}
\begin{block}{Human-intended dynamical system}
\[
\dot{x}_O^{H,*}=f_H(x_O,c;\theta_H)
\]
For a simple attractor model:
\[
\dot p_O^{H,*}=K_p(p_H^*-p_O),\qquad
\omega_O^{H,*}=f_R(R_O,R_H^*;K_R)
\]
$\theta_H=(p_H^*,R_H^*,K_p,K_R)$ represents desired pose and response aggressiveness.
\end{block}
\column{0.44\textwidth}
\begin{exampleblock}{Why it is attractive}
No assumption that the human solves a complete optimal-control problem; parameters can be updated from local motion and interaction-wrench evidence.
\end{exampleblock}
\begin{block}{Difference from Option B}
Option B explains behavior as approximate horizon optimization of a cost. Option C directly models local desired velocity; it is lighter-weight but less expressive about long-term tradeoffs.
\end{block}
\end{columns}
\end{frame}

\begin{frame}{Option C: Training and online inference}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Training without parameter labels}
Fit the likelihood of observed object motion using constrained system identification. The desired pose and response gains are latent variables rather than manually labeled quantities.
\end{block}
\column{0.48\textwidth}
\begin{block}{Switching between local models}
Use Expectation--Maximization (EM) offline or a particle filter online to infer $\theta_H$ and the active mode from motion and wrench measurements.
\end{block}
\end{columns}
\vspace{0.2cm}
\begin{alertblock}{Physical constraints}
Constrain $f_H$ so the predicted motion remains stable, feasible for the shared object, and consistent with contact and actuation limits.
\end{alertblock}
\vspace{0.12cm}
Related: constraint-aware dynamical-system intent estimation with particle filtering (Shao et al., 2024).
\end{frame}

\begin{frame}{Option D predicts responses to candidate robot actions}
\begin{columns}[c,onlytextwidth]
\column{0.43\textwidth}
\begin{block}{Passive prediction}
\[
p(u^H_{t:t+H}\mid h_t,c)
\]
\centering What will the human do next?
\end{block}
\column{0.08\textwidth}
\centering\Large\textcolor{orange}{$\Longrightarrow$}
\column{0.49\textwidth}
\begin{alertblock}{Interactive response model}
\[
p_\psi(u^H_{t:t+H}\mid h_t,c,u^R_{t:t+H})
\]
\centering How will the human respond if the robot executes this candidate action?
\end{alertblock}
\end{columns}
\vspace{0.28cm}
\begin{itemize}
  \item $u^H$: future interaction-wrench/action sequence; $u^R$: candidate robot action sequence.
  \item A diffusion/sequence model can represent multiple responses and role switches.
  \item It is predictive, but intention remains implicit unless the model also exposes a goal, trajectory, or preference variable.
\end{itemize}
\end{frame}

\begin{frame}[shrink=4]{Option D versus Diffusion Co-Policy}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Diffusion Co-Policy}
\[
p_\psi(u_{t:t+H}^H,u_{t:t+H}^R\mid h_t,c)
\]
\begin{itemize}
  \item jointly generates human and robot action sequences;
  \item directly imitates collaborative demonstrations;
  \item captures multimodality and role switching.
\end{itemize}
\end{block}
\column{0.48\textwidth}
\begin{block}{Human response model}
\[
p_\psi(u_{t:t+H}^H\mid h_t,c,\tilde u_{t:t+H}^R,z_H)
\]
\begin{itemize}
  \item evaluates each candidate robot action $\tilde u^R$;
  \item keeps planning and safety modular;
  \item can expose an intent variable $z_H$.
\end{itemize}
\end{block}
\end{columns}
\vspace{0.18cm}
\begin{alertblock}{Tradeoff}
The response model supports counterfactual planning, but needs diverse robot-action coverage and calibrated out-of-distribution uncertainty.
\end{alertblock}
\end{frame}

\begin{frame}[shrink=4]{Each representation serves a different modeling layer}
\scriptsize
\centering
\renewcommand{\arraystretch}{1.28}
\begin{tabularx}{\textwidth}{>{\bfseries}p{2.5cm}XXXX}
\toprule
Representation & Control interface & Key strength & Main weakness & Best use \\
\midrule
Interaction primitive & Latent mode $z_H$ & Cross-task structure & Needs mode discovery & Task-phase inference \\
Desired trajectory & Reference $\xi_H^*$ & Direct and visual & Misses preferences and robot influence & Shared-object MPC \\
Cost / preferences & $J_H,\,w_H,\,\phi$ & Interpretable and game-compatible & Identifiability and feature design & Negotiation and tradeoffs \\
Dynamical system & $f_H(x;\theta_H)$ & Lightweight online inference & Limited global multimodality & Local coordination \\
Response distribution & $p(u_H\mid u_R)$ & Flexible and interactive & Intention may remain implicit & Candidate-action evaluation \\
\bottomrule
\end{tabularx}
\vspace{0.32cm}
\begin{alertblock}{Proposed position}
Use structured intention variables as the control interface, while a learned response model captures multimodality, bounded rationality, and model mismatch.
\end{alertblock}
\end{frame}

\begin{frame}{Offline priors support online personalization}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Offline: learn shared structure}
\begin{itemize}
  \item task/interaction encoders from multi-person data;
  \item trajectory modes and residual cost features;
  \item robot-action-conditioned response model;
  \item calibrated uncertainty on held-out people.
\end{itemize}
\end{block}
\column{0.48\textwidth}
\begin{block}{Online: update compact variables}
\begin{itemize}
  \item belief over intent mode $b_t(z_H)$;
  \item desired motion parameters $\xi_H^*,\theta_H$;
  \item preference weights $w_H$;
  \item optional small person-specific adapter.
\end{itemize}
\end{block}
\end{columns}
\vspace{0.22cm}
\begin{alertblock}{Update from physical interaction}
\[
b_{t+1}(z_H,\theta_H,w_H)\propto
p(\mathbf{w}_t^{\mathrm{int,obs}},x_{t+1}^O\mid h_t,c,u_t^R,z_H,\theta_H,w_H)\,b_t(z_H,\theta_H,w_H)
\]
\end{alertblock}
\end{frame}

\begin{frame}[shrink=5]{Data collection supplies weak and self-supervision}
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Minimum synchronized modalities}
\begin{itemize}
  \item object pose, velocity, and acceleration;
  \item 6D interaction wrench;
  \item robot state and commanded action;
  \item task goal, geometry, contact configuration;
  \item optional human body/arm pose and vision.
\end{itemize}
\end{block}
\column{0.48\textwidth}
\begin{block}{Where training signals come from}
\begin{itemize}
  \item \textbf{weak labels}: assigned goal, path condition, task phase, role instruction;
  \item \textbf{self-supervision}: future object motion and interaction wrench;
  \item \textbf{preference evidence}: choices under controlled effort/force/path tradeoffs;
  \item \textbf{latent learning}: EM / IOC likelihood when no direct label exists.
\end{itemize}
\end{block}
\end{columns}
\vspace{0.2cm}
\begin{alertblock}{Practical collection plan}
Use accessible human--human demonstrations for pretraining, then collect human--robot trials with diverse robot actions for interactive-response identification. A curated multimodal benchmark can itself support a dataset/benchmark paper.
\end{alertblock}
\end{frame}

\begin{frame}{Separate model quality from team performance}

Motion vs. motion+wrench; passive vs. robot-action-conditioned; fixed vs. learned features; population prior vs. online personalization.

\begin{frame}{SP1: Build the physical evidence base}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.49\textwidth}
\begin{block}{Theme and focus}
\textbf{Physically grounded multimodal data}
\begin{itemize}
  \item Build shared-object transport experiments
  \item Synchronize motion, object state, and force/torque

  \item Compare motion-only, wrench-only, and multimodal inference

\end{block}
\column{0.47\textwidth}
\begin{alertblock}{Challenges $\rightarrow$ solutions}
\begin{itemize}
  \item No direct intent labels $\rightarrow$ hierarchical task/strategy/action labels
  \item Noisy contact data $\rightarrow$ filtering and contact-state estimation
  \item High person variability $\rightarrow$ probabilistic latent-state models
  \item Costly data $\rightarrow$ human--human, simulation, and HRI data mixture
\end{itemize}
\end{alertblock}
\end{columns}
\vspace{0.18cm}
\paperout{A multimodal benchmark and systematic study of force/torque for human intent inference.}

\begin{frame}[shrink=6]{SP2: Make intent and preference explicit}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Bounded-rational human model}
\[
J_H=\sum_t {w_t^H}^{\!\top}\phi_\psi(x_t,u_t^H,u_t^R,c)
\]
Reusable features may represent progress, pose, effort, interaction force, safety, smoothness, and coordination.
\begin{itemize}
  \item Infer intent, role, preference, and rationality online
  \item Represent the human's belief about robot intent
  \item Treat the human as another adapting learner
\end{itemize}
\end{block}
\column{0.48\textwidth}
\begin{alertblock}{Challenges $\rightarrow$ solutions}
\begin{itemize}
  \item Cost identifiability $\rightarrow$ Bayesian / maximum-entropy IOC
  \item Ambiguous intent $\rightarrow$ informative robot actions
  \item Unknown adaptation $\rightarrow$ latent learning-state estimation
  \item Online computation $\rightarrow$ local game approximations and MPC
\end{itemize}
\end{alertblock}
\end{columns}
\vspace{0.15cm}
\paperout{Peer-aware intent and preference inference for physical co-manipulation.}
\end{frame}

\begin{frame}[shrink=8]{SP3: Learn responses to possible robot actions}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.49\textwidth}
\begin{block}{Interactive response prediction}
\[
p_\psi(u^H_{\mathrm{future}}\mid h_t,c,u^R_{\mathrm{future}})
\]
\begin{itemize}
  \item Predict multimodal human response trajectories
  \item Condition on motion, wrench, context, and candidate robot actions
  \item Predict uncertainty and latent-state transitions
  \item Compare Transformer, diffusion, and latent-variable models
\end{itemize}
\end{block}
\column{0.47\textwidth}
\begin{alertblock}{Challenges $\rightarrow$ solutions}
\begin{itemize}
  \item Limited action coverage $\rightarrow$ diverse robot probing
  \item Correlation vs. response $\rightarrow$ counterfactual conditioning
  \item OOD actions $\rightarrow$ uncertainty estimation and detection
  \item Slow inference $\rightarrow$ distilled or few-step diffusion
\end{itemize}
\end{alertblock}
\end{columns}
\vspace{0.15cm}
\paperout{Robot-action-conditioned diffusion models for multimodal human response prediction.}
\end{frame}

\begin{frame}{SP4: Close the loop through safe control}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Human-model-aware planning}
\[
u_R^*=\arg\min_{u_R}\;\mathbb{E}_{u_H\sim p_\psi}[J_R]
\]
subject to object dynamics, actuation, force, collision, and safety constraints.
\begin{itemize}
  \item Optimize progress, team effort, and interaction force
  \item Replan as the human model is updated
  \item Use a safety filter before execution
\end{itemize}
\end{block}
\column{0.48\textwidth}
\begin{alertblock}{Challenges $\rightarrow$ solutions}
\begin{itemize}
  \item Prediction error $\rightarrow$ short-horizon MPC
  \item Distributional complexity $\rightarrow$ scenario sampling
  \item Excess conservatism $\rightarrow$ risk-sensitive objectives
  \item Hard safety limits $\rightarrow$ CBF / constrained safety filter
\end{itemize}
\end{alertblock}
\end{columns}
\vspace{0.15cm}
\paperout{Uncertainty-aware planning and safe control with interactive human response models.}
\end{frame}

\begin{frame}[shrink=7]{SP5: Transfer human concepts across tasks and people}
\footnotesize
\begin{columns}[T,onlytextwidth]
\column{0.48\textwidth}
\begin{block}{Task-conditioned generalization}
Shared human latent concepts:
\begin{itemize}
  \item intent and local strategy;
  \item leader--follower role and compliance;
  \item effort, safety, and interaction preferences;
  \item reaction delay and adaptation state.
\end{itemize}
Encode goals, object dynamics, geometry, contacts, and constraints as task context. Adapt only latent states, preference weights, or a small adapter online.
\end{block}
\column{0.48\textwidth}
\begin{alertblock}{Challenges $\rightarrow$ solutions}
\begin{itemize}
  \item Different task structures $\rightarrow$ object-centric states and wrenches
  \item Limited physical data $\rightarrow$ simulation and human--human pretraining
  \item Negative transfer $\rightarrow$ structured task encoders
  \item Unstable personalization $\rightarrow$ Bayesian updates or small adapters
\end{itemize}
\end{alertblock}
\end{columns}
\vspace{0.15cm}
\paperout{A task-conditioned human model for cross-task co-manipulation and online personalization.}
\end{frame}
